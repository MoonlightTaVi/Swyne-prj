// Шаблоны
ОБЪЕКТ-ПОЯСНЕНИЕ: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ()[п+,ч,р] = ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<>, ПРИЛМЕСТ<>
ОДНОРОДНОЕ-ОДНОРОДНОЕ: ОПРЕДЕЛЕНИЕ()[п,р,ч] = ОПРЕДЕЛЕНИЕ<>, ОПРЕДЕЛЕНИЕ<>
ОДНОРОДНОЕ-СОЮЗ: ОПРЕДЕЛЕНИЕ()[] = СОЮЗ<> ОПРЕДЕЛЕНИЕ<>
ОДНОРОДНОЕ-ОДНОРОДНОЕ: ОПРЕДЕЛЕНИЕ()[п,р,ч] = ОПРЕДЕЛЕНИЕ<> ОПРЕДЕЛЕНИЕ<>
СРАВН-ЧЕМ: ОПРЕДЕЛЕНИЕ()[] = ОПРЕДЕЛЕНИЕ<сравн> ЧИСЛО<>
ОБЪЕКТ-КАЧЕСТВО: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ()[п+,р,ч] = ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<> ОПРЕДЕЛЕНИЕ<> ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<>
ОБЪЕКТ-КАЧЕСТВО: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ()[п+,р,ч] = ОПРЕДЕЛЕНИЕ<> ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<>
ОБЪЕКТ-КАЧЕСТВО: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ()[п+,р,ч] = ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<> ОПРЕДЕЛЕНИЕ<>
ОБЪЕКТ-ПРЕДЛ: ДОПОЛНЕНИЕ()[] = ПРЕДЛ<> ДОПОЛНЕНИЕ<>
ОБЪЕКТ-ПРЕДЛ: ЧИСЛО()[] = ПРЕДЛ<> ЧИСЛО<>
ОБЪЕКТ-СРАВН: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ()[] = ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<> ОПРЕДЕЛЕНИЕ<сравн>
ОБЪЕКТ-КАЧЕСТВО: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ()[] = ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<> ОПРЕДЕЛЕНИЕ<>
//ПРИНАДЛЕЖАЩЕЕ-ВЛАДЕЛЕЦ: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ(!рд)[] = ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<!рд> ДОПОЛНЕНИЕ<рд>
//ПРИНАДЛЕЖАЩЕЕ-ВЛАДЕЛЕЦ: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ(!рд)[] = ДОПОЛНЕНИЕ<рд> ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<!рд>
//ПРИНАДЛЕЖАЩЕЕ-ВЛАДЕЛЕЦ: ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ()[] = ДОПОЛНЕНИЕ<рд> ДОПОЛНЕНИЕ<рд>
ДЕЙСТВИЕ-АРГУМЕНТ: ИНФИНИТИВ()[] = ИНФИНИТИВ<!Г> ДОПОЛНЕНИЕ<>
ДЕЙСТВИЕ-АРГУМЕНТ: ИНФИНИТИВ()[] = ИНФИНИТИВ<!Г> ОПРЕДЕЛЕНИЕ<>
ПРЕДШЕСТВУЕТ-СЛЕДУЕТ: ИНФИНИТИВ()[ОСН] = ИНФИНИТИВ<>, ИНФИНИТИВ<>
ОСН-ИНФ: СКАЗУЕМОЕ()[] = СКАЗУЕМОЕ<> ИНФИНИТИВ<>
ДЕЙСТВИЕ-АКТЁР: СКАЗУЕМОЕ()[ч] = ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<> СКАЗУЕМОЕ<>
ДЕЙСТВИЕ-АРГУМЕНТ: СКАЗУЕМОЕ()[] = СКАЗУЕМОЕ<> ПОДЛЕЖАЩЕЕилиДОПОЛНЕНИЕ<>
ДЕЙСТВИЕ-АРГУМЕНТ: СКАЗУЕМОЕ()[] = СКАЗУЕМОЕ<> ОПРЕДЕЛЕНИЕ<>
ДЕЙСТВИЕ-ЧИСЛОВОЙАРГУМЕНТ: СКАЗУЕМОЕ()[] = СКАЗУЕМОЕ<> ЧИСЛО<>
